// frontend/assets/auth.js import { api, getToken, setToken } from "/assets/api.js"; function isAuthPage() { const p = location.pathname; return p.endsWith("/login.html") || p.endsWith("/criar-conta.html"); } function addTokenToLinks() { const t = getToken(); if (!t) return; // adiciona token só em links internos (mesmo domínio) document.querySelectorAll('a[href^="/"]').forEach((a) => { try { const u = new URL(a.getAttribute("href"), location.origin); if (!u.searchParams.get("t")) u.searchParams.set("t", t); a.setAttribute("href", u.pathname + u.search + u.hash); } catch {} }); } /** * ========================= * ✅ NOVO: Auth API helpers * ========================= * Mantém compatibilidade mesmo se o backend tiver endpoint diferente: * tenta /api/auth/register e fallback /api/auth/signup */ async function postAuth(path, body) { try { return await api(path, { method: "POST", body }); } catch (e) { // fallback se o backend não tiver essa rota if (String(e?.message || "").includes("404") || String(e?.message || "").includes("Rota não encontrada")) { throw e; } throw e; } } /** * ✅ NOVO: criar conta * Espera que o backend retorne { token } ou { ok, token }. */ export async function register(name, email, password, referrerCode = null) { const payload = { name, email, password, referrerCode: referrerCode || null, }; // tenta rota mais comum let data; try { data = await api("/api/auth/register", { method: "POST", body: payload }); } catch (e1) { // fallback comum data = await api("/api/auth/signup", { method: "POST", body: payload }); } const token = data?.token || data?.data?.token || data?.jwt || null; if (token) { setToken(token); addTokenToLinks(); } return data; } /** * ✅ NOVO: login */ export async function login(email, password) { const payload = { email, password }; let data; try { data = await api("/api/auth/login", { method: "POST", body: payload }); } catch (e1) { // fallback comum data = await api("/api/auth/signin", { method: "POST", body: payload }); } const token = data?.token || data?.data?.token || data?.jwt || null; if (token) { setToken(token); addTokenToLinks(); } return data; } /** * ✅ NOVO: logout local (client-side) */ export function logout() { setToken(null); } /** * ========================= * ✅ Mantido: ensureLoggedIn * ========================= */ export async function ensureLoggedIn({ redirectToLogin = true, requireActive = false, redirectToAssinatura = true, } = {}) { // ✅ evita loop infinito: não forçar auth dentro do login/criar-conta if (isAuthPage()) return null; const token = getToken(); // sem token: redireciona if (!token) { if (redirectToLogin) { const next = encodeURIComponent(location.pathname + location.search); location.href = `/login.html?next=${next}`; } return null; } try { const me = await api("/api/me"); addTokenToLinks(); // ✅ se quiser exigir assinatura ativa em páginas do painel const sub = me?.user?.subscription || me?.subscription || null; const status = sub?.status || "inactive"; if (requireActive && status !== "active") { if (redirectToAssinatura) { location.href = "/assinatura.html"; } return me; } return me; } catch (e) { // token inválido/expirado setToken(null); if (redirectToLogin) { const next = encodeURIComponent(location.pathname + location.search); location.href = `/login.html?next=${next}`; } return null; } } // ✅ roda automaticamente em páginas que incluem auth.js ensureLoggedIn().catch(() => {});
